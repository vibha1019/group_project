    function setRandomPosition(carrot) {
        const randomX = Math.floor(Math.random() * window.innerWidth);
        const randomY = -30; // Starts above the top of the screen
        carrot.style.left = `${randomX}px`;
        carrot.style.top = `${randomY}px`;
    }
    // Function to make a carrot fall
    function fall(carrot) {
        const speed = Math.floor(Math.random() * 5) + 3; // Random speed between 3 and 7
        const interval = setInterval(function () {
            const currentTop = parseInt(carrot.style.top, 10);
            const newTop = currentTop + speed;
            carrot.style.top = `${newTop}px`;
            // Check if the carrot has reached the bottom of the screen
            if (newTop > window.innerHeight) {
                clearInterval(interval);
                setRandomPosition(carrot); // Reset position for the next fall
                fall(carrot); // Start falling again
            }
        }, 30); // Update every 30 milliseconds
    }
    // Create and start falling for multiple carrots
    const numberOfCarrots = 10;
    for (let i = 0; i < numberOfCarrots; i++) {
        const carrot = createCarrot();
        setRandomPosition(carrot);
        fall(carrot);
    });
